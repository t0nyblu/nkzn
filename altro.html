//********************* // Aggiungi evento click per mostrare solo l'elemento cliccato listItem.addEventListener('click', function () { var ul = document.getElementById("glossaryList"); var li = ul.getElementsByTagName('li'); for (var j = 0; j < li.length; j++) { if (li[j] !== this) { li[j].style.display = "none"; // Nascondi gli altri elementi } else { li[j].style.display = ""; // Mostra solo l'elemento cliccato } } });
